def MetodoGradienteConjugadoCuadratico(A, b, x0, eps=0.01):
    #Escogemos estimacion x0 de h y calculamos r0 y p0
    n = len(b)
    r = [b - A*x0] + [0]*n
    p = [r[0]] + [0]*n
    a = [0] * (n+1)
    beta = [0] * (n+1)
    x = [x0] + [0]*n
    for i in [0..n]:
        if (p[i].norm() < eps):
            break
        a[i] = (p[i].dot_product(r[i])) / (p[i].dot_product(A*p[i]))
        x[i+1] = x[i] + a[i]*p[i]
        r[i+1] = r[i] - a[i]*A*p[i]
        beta[i] = (r[i+1].dot_product(A*p[i])) / (p[i].dot_product(A*p[i]))
        p[i+1] = r[i+1]-beta[i]*p[i]
    return x[i], x[:i+1]

def MetodoGradienteConjugado(f, x0, eps=0.01):
    # Este metodo es la generalizacion para aproximar cualquier ecuacion a una cuadratica, 
    # de forma que el metodo funcione correctamente
    A = f.hessian().subs({v: 0 for v in f.variables()})
    # Cuidado porque el signo de b cambia respecto a la formula de la forma cuadratica
    b = -f.gradient().subs({v: 0 for v in f.variables()})
    return MetodoGradienteConjugadoCuadratico(A,b,x0,eps)

def MetodoGradienteConjugadoNoLineal(f, x0, eps=0.01):
    #Escogemos estimacion x0 de h y calculamos r0 y p0
    n = len(x0)
    # df(x+ap) / da
    # Taylor aprox como polinomio de grada
    df = f.diff()
    A = [f.hessian()(*x0)]
    r = [-df(x0)]
    p = [r[0]]
    a = [0] * (n+1)
    beta = [0] * (n+1)
    x = [x0]
    for i in [0..n]:
        if (p[i].norm() < eps):
            break
        a[i] = (-df(x[i])*p[i]) / (p[i].dot_product(A[i]*p[i]))
        x[i+1] = x[i] + a[i]*p[i]
        r[i+1] = -f(x[i+1])
        p[i+1] = r[i+1]-beta[i]*p[i]
        A[i+1] = A[i] + f.hessian()(*x[i])
        bPR = (r[i+1]*(r[i+1]-r[i]) / (r[i]*r[i])
        if bPR > 0:
            beta[i] = bPR
        else:
            beta[i] = 0
            
    return x[i], x[:i+1]

def MetodoGradienteConjugadoNoLineal(f, x0, line_search, beta_method, eps=0.0001, nmax=1000):
    df = f.diff()
    betas = [None]
    r = [-df(*x0)]
    p = r[0]
    x = [x0]
    for i in [0..nmax]:
        a = line_search(f, df, x[i], p)
        x.append(x[i] + a * p)
        grad = df(*x[i+1])
        r.append(-grad)
        if (r[i+1].norm() < eps):
            break
        beta = beta_method(r[i], r[i+1])
        betas.append(beta)
        p = beta*p - grad
    return x[i], x[:i+1], betas

def beta_fletcher_reeves(r_prev, r):
    return r.norm()^2 / r_prev.norm()^2

def beta_polak_riviere(r_prev, r):
    return r.dot_product(r-r_prev) / r_prev.norm()**2

def line_search_wolfe(f, grad, x, p, sigma1=0.0001, sigma2=0.2):
    # Usamos el line_search implementado en scipy
    import numpy as np
    from scipy.optimize import line_search

    def func(x): # Objective function
        return np.float64(f(*x))

    def Df(x):
        return np.array(f.diff()(*x), dtype=np.float64)
    start_point=np.array([2., -1.8])
    delta = np.array(p, dtype=np.float64)
    s = line_search(
        f=func,
        myfprime=Df,
        xk=x, pk=delta, c1=sigma1, c2=sigma2)[0]
    return float(s)
